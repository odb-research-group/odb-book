\begin{lstlisting}
; 2nodes.cpp

#include <iostream>

#include "generic.h"
#include "odb.h"

auto oOdb = odb::COdb();

int main()
    {
    // 2 people
    oOdb.MakeNode("Ina");
    oOdb.MakeNode("Rob");
    
    // a relation
    oOdb.MakeReason("loves");
    
    // binding
    oOdb.LinkNode2Node("Ina", "loves", "Rob");
    
    // show us
    std::cout << "---------------- all nodes" << '\n';
    for ( auto const & a:oOdb.Nodes() )
        {
        std::cout << *a << '\n';
        }
    }
\end{lstlisting}
